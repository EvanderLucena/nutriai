// API Response shapes for Food catalog

→ { success: true, data: { content: FoodResponse[], page: int, size: int, total: int } }

PATCH /api/v1/foods/{id} → 200 { success: true, data: FoodResponse }
DELETE /api/v1/foods/{id} → 204

// FoodResponse fields
{ id, type, name, category, per100Kcal, per100Prot, per100Carb, per100Fat, per100Fiber,
  presetGrams, presetKcal, presetProt, presetCarb, presetFat, portionLabel,
  portions: [{ id, name, grams }], usedCount, createdAt, updatedAt }

// API Response shapes for Plan
GET /api/v1/patients/{id}/plan → 200 PlanResponse
PATCH /api/v1/patients/{id}/plan → 200 PlanResponse
POST /api/v1/patients/{id}/plan/meals → 201 MealSlotResponse
PATCH /api/v1/patients/{id}/plan/meals/{mealId} → 200 MealSlotResponse
DELETE /api/v1/patients/{id}/plan/meals/{mealId} → 204
POST /api/v1/patients/{id}/plan/meals/{mealId}/options → 201 MealOptionResponse
PATCH /api/v1/patients/{id}/plan/meals/{mealId}/options/{optionId} → 200 MealOptionResponse
DELETE /api/v1/patients/{id}/plan/meals/{mealId}/options/{optionId} → 204
POST /api/v1/patients/{id}/plan/meals/{mealId}/options/{optionId}/items → 201 MealFoodResponse
PATCH /api/v1/patients/{id}/plan/meals/{mealId}/options/{optionId}/items/{itemId} → 200 MealFoodResponse
DELETE /api/v1/patients/{id}/plan/meals/{mealId}/options/{optionId}/items/{itemId} → 204
POST /api/v1/patients/{id}/plan/extras → 201 ExtraResponse
PATCH /api/v1/patients/{id}/plan/extras/{extraId} → 200 ExtraResponse
DELETE /api/v1/patients/{id}/plan/extras/{extraId} → 204

// PlanResponse structure
{ id, episodeId, title, notes, kcalTarget, protTarget, carbTarget, fatTarget,
  meals: [{ id, label, time, options: [{ id, name, items: [{ id, foodId, foodName, qty, grams, prep, kcal, prot, carb, fat }] }] }],

  createdAt, updatedAt }
